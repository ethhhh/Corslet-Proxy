// how it works

    
Three steps. Done.

    
No packages, no installs, no tokens. Just a URL transformation.

    

      

        
01 —

        
Prefix your URL

        
Prepend https://corslet.ethh.workers.dev/?url= to any target URL.

      

      

        
02 —

        
We proxy it

        
corslet forwards your full request — method, headers, body — to the target server from our Cloudflare edge.

      

      

        
03 —

        
CORS headers added

        
The response is returned with Access-Control-Allow-Origin: * so your browser is happy.

// ✓ One-line fix for CORS errors
const PROXY = 'https://corslet.ethh.workers.dev/?url=';

const res = await fetch(PROXY + 'https://api.example.com/data');
const data = await res.json();

// With custom headers
const res2 = await fetch(PROXY + 'https://api.example.com/secure', {
  headers: { 'Authorization': 'Bearer your-token' }
});


        
import axios from 'axios';

const PROXY = 'https://corslet.ethh.workers.dev/?url=';

// GET
const { data } = await axios.get(PROXY + 'https://api.example.com/data');

// POST
await axios.post(PROXY + 'https://api.example.com/submit', {
  name: 'hello'
});


        
curl "https://corslet.ethh.workers.dev/?url=https://api.example.com/data"

# With authorization header
curl -H "Authorization: Bearer token" \
  "https://corslet.ethh.workers.dev/?url=https://api.example.com/secure"

# POST with body
curl -X POST \
  -H "Content-Type: application/json" \
  -d '{"key":"value"}' \
  "https://corslet.ethh.workers.dev/?url=https://api.example.com/data"

  }
});
